经典的[两数之和](https://leetcode-cn.com/problems/two-sum/)~

```js
//JS
var twoSum = function(nums, target) {
    for(let i = 0; i < nums.length; i++){
        for(let j = i + 1; j < nums.length; j++){
            if(nums[i] + nums[j] == target){
                return [i, j];
                //return new int[]{i, j}; JAVA代码
                //return {i, j}; C++代码
            }
        }
    }
    return false;
};
```
